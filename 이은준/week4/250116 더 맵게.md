import java.util.*;

class Solution {
    public int solution(int[] scoville, int K) {
        PriorityQueue<Integer> scovilles = new PriorityQueue<>();

        for (int level: scoville) {
            scovilles.add(level);
        }
        
        int mix = -1;
        int answer = 0;
        
        while (scovilles.peek() < K) {        
            int first = scovilles.poll();
            int second = scovilles.poll();
            mix = first + second * 2;
            scovilles.add(mix);
            answer++;
        }
