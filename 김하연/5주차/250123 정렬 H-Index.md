import java.util.Arrays;

class Solution {
    public int solution(int[] citations) {

        int len = citations.length-1;
        Arrays.sort(citations);

        if (citations[len]==0) return 0;

        int answer = 0;
        for (int i=len; i>=0; i--) {
            if (citations[i]<=answer) {
                break;
            }

            answer++;
        }

        return answer;
    }
}
