import java.util.Stack;

class Solution {
    boolean solution(String s) {
        Stack<String> parentheses = new Stack<>();
        
        for (char c : s.toCharArray()) {
            if (c=='(') {
                parentheses.push("(");
            } else if (c==')') {
                if (!parentheses.isEmpty()) {
                    parentheses.pop();
                } else return false;
            }
        }
        
        return parentheses.isEmpty();
    }
}
