import java.util.*;

public class Solution {
    public int[] solution(int []arr) {
        List<Integer> list = new ArrayList<>();
        for (int i : arr) {
            if (!list.contains(i)) list.add(i);
        }

        return list.stream().mapToInt(Integer::intValue).toArray();
    }
}
