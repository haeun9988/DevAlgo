class Solution {
    public int[] solution(int[] prices) {
        int length = prices.length;

        int[] result = new int[prices.length];

        for (int i=0; i<length; i++) {
            for (int j=i+1; j<length; j++) {
                result[i]++;
                if (prices[j]<prices[i]) break;
            }
        }

        return result;
    }
}
