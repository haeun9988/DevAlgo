import java.util.PriorityQueue;

class Solution {
    public int solution(int[] scoville, int K) {
        PriorityQueue<Integer> sc = new PriorityQueue<>();

        for (int i: scoville) {
            sc.offer(i);
        }

        int answer = 0;

        while (sc.peek()<K) {
            if (sc.size()<2) return -1;

            int s1 = sc.poll();
            int s2 = sc.poll();

            sc.offer(s1+s2*2);
            answer++;
        }

        return answer;
    }
}
