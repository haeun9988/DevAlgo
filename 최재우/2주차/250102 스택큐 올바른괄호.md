import java.util.*;
class Solution {
boolean solution(String s) {
        Deque<Character> deque = new ArrayDeque<>();
        char[] charArray = s.toCharArray();
        for (Character c : charArray) {
            if (c == '(') {
                deque.addLast(c);
            }
            if (c == ')') {
                if (deque.isEmpty()) {
                    return false;
                } else {
                    deque.pollLast();
                }
            }
        }
        /**
