import java.util.*;
class Solution {
       public int solution(int[] citations) {
        int n = citations.length;
        Arrays.sort(citations);
        System.out.println(Arrays.toString(citations));

        int hMax = 0;
        for (int i = 0; i < citations.length; i++) {

            int h = n - i; //5 -> 4 -> 3 -> 2 -> 1

            if (citations[i] >= h) { // 0>=5 -> 1>=4 -> 3>=3
                hMax = h;
                break;
            }
        }
        return hMax;
    }
}
