import java.util.*;
class Solution {
public int solution(int[] scoville, int K) throws InterruptedException {
        int count = 0;

        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        for (int s : scoville) {
            minHeap.add(s);
        }

        while (minHeap.peek() < K) {
            if (minHeap.size() == 1) {
                return -1;
            }
            minHeap.add(minHeap.poll() + minHeap.poll() * 2);
            count++;
        }

        return count;
    }
}
